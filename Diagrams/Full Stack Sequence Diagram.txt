@startuml
autonumber

actor Researcher
participant "Flask App" as UI
participant "MitID" as MitID
participant "Session" as S
participant "LPR3 Registry" as DB
participant "Audit Log" as Audit

Researcher -> UI : Open app
UI --> Researcher : Show login page
Researcher -> UI : Start login
UI -> MitID : Authenticate researcher

alt Login successful
    MitID --> UI : Identity verified

    create S
    UI -> S : Create session
    activate S

    UI --> Researcher : Show search page

    loop Until valid query submitted
        Researcher -> UI : Enter search criteria
        UI -> MitID : Check authorization

        alt Access granted
            MitID --> UI : Permission granted
            UI -> DB : Execute query

            alt Results found
                DB --> UI : Return matching records
                UI -> Audit : Log successful query
                UI --> Researcher : Display results

                opt Export data
                    Researcher -> UI : Request export
                    UI -> MitID : Check export permission

                    alt Export allowed
                        MitID --> UI : Export approved
                        UI -> Audit : Log export action
                        UI --> Researcher : Download file
                    else Export denied
                        MitID --> UI : Export denied
                        UI -> Audit : Log denied export
                        UI --> Researcher : Show export denied
                    end
                end

            else No results found
                DB --> UI : Empty result set
                UI -> Audit : Log empty query
                UI --> Researcher : Show "No results found"

            else Query failed
                DB --> UI : Error message
                UI -> Audit : Log failed query
                UI --> Researcher : Show query error
            end

        else Access denied
            MitID --> UI : Permission denied
            UI -> Audit : Log unauthorized attempt
            UI --> Researcher : Show access denied
        end
    end

    opt Logout
        Researcher -> UI : Logout
        UI -> S : Destroy session
        destroy S
        UI --> Researcher : Session ended
    end

else Login failed
    MitID --> UI : Authentication failed
    UI -> Audit : Log failed login
    UI --> Researcher : Show login error
end

@enduml
