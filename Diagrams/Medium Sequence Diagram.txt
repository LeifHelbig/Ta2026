@startuml
autonumber

actor Researcher
participant "Flask App" as UI
participant "MitID" as MitID
participant "LPR3 Registry" as DB
participant "Audit Log" as Audit

Researcher -> UI : Open app
UI --> Researcher : Show login page

Researcher -> UI : Start login
UI -> MitID : Authenticate researcher

alt Login successful
    MitID --> UI : Identity verified
    UI --> Researcher : Show search page

    Researcher -> UI : Enter search criteria
    UI -> DB : Execute query
    DB --> UI : Return results

    UI ->> Audit : Log query
    UI --> Researcher : Display results

    opt Export data
        Researcher -> UI : Request export
        UI ->> Audit : Log export
        UI --> Researcher : Download file
    end

else Login failed
    MitID --> UI : Authentication failed
    UI ->> Audit : Log failed login
    UI --> Researcher : Show login error
end

@enduml
